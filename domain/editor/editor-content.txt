     1	//! File: domain/editor/editor_core/src/executor.rs
     2	//! Purpose: Command execution and history integration.
     3	
     4	use crate::{
     5		Command, CommandMetadata, CommandOutcome, HistoryEntry, HistoryStack, TransactionMetadata,
     6	};
     7	
     8	#[derive(Debug, Clone, PartialEq, Eq)]
     9	pub struct ExecutedCommand {
    10		pub metadata: CommandMetadata,
    11	}
    12	
    13	#[derive(Debug, Clone, PartialEq, Eq)]
    14	pub struct ExecutedTransaction {
    15		pub transaction: TransactionMetadata,
    16		pub commands: Vec<ExecutedCommand>,
    17	}
    18	
    19	impl ExecutedTransaction {
    20		pub fn into_history_entry(self) -> HistoryEntry {
    21			HistoryEntry::new(
    22				self.transaction,
    23				self.commands.into_iter().map(|cmd| cmd.metadata).collect(),
    24			)
    25		}
    26	}
    27	
    28	pub struct CommandExecutor;
    29	
    30	impl CommandExecutor {
    31		pub fn execute_command<C>(
    32			ctx: &mut C::Context<'_>,
    33			command: &mut C,
    34		) -> Result<Option<ExecutedCommand>, C::Error>
    35		where
    36			C: Command,
    37		{
    38			match command.apply(ctx)? {
    39				CommandOutcome::NoOp => Ok(None),
    40				CommandOutcome::Applied => Ok(Some(ExecutedCommand {
    41					metadata: command.metadata().clone(),
    42				})),
    43			}
    44		}
    45	
    46		pub fn execute_transaction<C>(
    47			ctx: &mut C::Context<'_>,
    48			transaction: TransactionMetadata,
    49			commands: &mut [C],
    50		) -> Result<ExecutedTransaction, C::Error>
    51		where
    52			C: Command,
    53		{
    54			let mut executed = Vec::new();
    55	
    56			for command in commands {
    57				if let Some(entry) = Self::execute_command(ctx, command)? {
    58					executed.push(entry);
    59				}
    60			}
    61	
    62			Ok(ExecutedTransaction {
    63				transaction,
    64				commands: executed,
    65			})
    66		}
    67	
    68		pub fn push_history(
    69			history: &mut HistoryStack,
    70			transaction: ExecutedTransaction,
    71		) {
    72			if transaction.commands.is_empty() {
    73				return;
    74			}
    75	
    76			history.push_applied(transaction.into_history_entry());
    77		}
    78	}

   129		stable_name: impl Into<String>,
   130		display_name: impl Into<String>,
   131		path: InspectorPath,
   132		value: ReflectValueRef<'_>,
   133	) -> InspectorField {
   134		let stable_name = stable_name.into();
   135		let display_name = display_name.into();
   136	
   137		if let Some(struct_ref) = value.struct_ref() {
   138			return build_struct_field(stable_name, display_name, path, struct_ref);
   139		}

   141		build_leaf_field(stable_name, display_name, path, value)
   142	}

   144	/// File: domain/editor/editor_inspector/src/bridge/ecs_adapter.rs

   146	fn build_struct_field(
   147		stable_name: String,
   148		display_name: String,
   149		path: InspectorPath,
   150		struct_ref: StructValueRef<'_>,
   151	) -> InspectorField {
   152		let mut field = InspectorField::new(
   153			stable_name,
   154			display_name,
   155			path.clone(),
   156			InspectorValue::Group,
   157		)
   158			.read_only(true);
   159	
   160		for meta in struct_ref.fields() {
   161			let child_path = path.child_field(meta.name);
   162			let Some(child_value) = struct_ref.field(meta.name) else {
   163				continue;
   164			};
   165	
   166			field = field.with_child(build_reflect_field(
   167				meta.name,
   168				meta.display_name,
   169				child_path,
   170				child_value,
   171			));
   172		}
   173	
   174		field
   175	}

   100	fn set_value_at_path(
   101		current: ReflectValueMut<'_>,
   102		segments: &[InspectorPathSegment],
   103		value: &InspectorEditValue,
   104	) -> Result<(), InspectorEditError> {
   105		match segments.split_first() {
   106			None => apply_primitive_edit(current, value),
   107			Some((segment, rest)) => match segment {
   108				InspectorPathSegment::Field(name) => {
   109					let mut struct_mut = current
   110						.struct_mut()
   111						.ok_or(InspectorEditError::InvalidPath)?;
   112	
   113					let child = struct_mut
   114						.field_mut(name)
   115						.ok_or(InspectorEditError::InvalidPath)?;
   116	
   117					set_value_at_path(child, rest, value)
   118				}
   119				InspectorPathSegment::Index(_) => Err(InspectorEditError::UnsupportedPathSegment),
   120			},
   121		}
   122	}
   123	
   124	fn apply_primitive_edit(
   125		mut current: ReflectValueMut<'_>,
   126		value: &InspectorEditValue,
   127	) -> Result<(), InspectorEditError> {
   128		let actual_type = current.type_info().stable_name.to_string();
   129	
   130		match value {
   131			InspectorEditValue::Bool(next) => {
   132				let slot = current
   133					.downcast_mut::<bool>()
   134					.ok_or_else(|| InspectorEditError::UnsupportedValueType {
   135						actual_type: actual_type.clone(),
   136					})?;
   137				*slot = *next;
   138				Ok(())
   139			}
   140			InspectorEditValue::Integer(next) => {
   141				if let Some(slot) = current.downcast_mut::<i8>() {
   142					*slot = i8::try_from(*next).map_err(|_| InspectorEditError::IntegerOutOfRange {
   143						target_type: "i8",
   144						value: *next,
   145					})?;
   146					return Ok(());
   147				}
   148				if let Some(slot) = current.downcast_mut::<i16>() {
   149					*slot = i16::try_from(*next).map_err(|_| InspectorEditError::IntegerOutOfRange {
   150						target_type: "i16",
   151						value: *next,
   152					})?;
   153					return Ok(());
   154				}
   155				if let Some(slot) = current.downcast_mut::<i32>() {
   156					*slot = i32::try_from(*next).map_err(|_| InspectorEditError::IntegerOutOfRange {
   157						target_type: "i32",
   158						value: *next,
   159					})?;
   160					return Ok(());
   161				}
   162				if let Some(slot) = current.downcast_mut::<i64>() {
   163					*slot = *next;
   164					return Ok(());
   165				}
   166				if let Some(slot) = current.downcast_mut::<isize>() {
   167					*slot =
   168						isize::try_from(*next).map_err(|_| InspectorEditError::IntegerOutOfRange {
   169							target_type: "isize",
   170							value: *next,
   171						})?;
   172					return Ok(());
   173				}
   174				if let Some(slot) = current.downcast_mut::<u8>() {
   175					*slot = u8::try_from(*next).map_err(|_| InspectorEditError::IntegerOutOfRange {
   176						target_type: "u8",
   177						value: *next,
   178					})?;
   179					return Ok(());
   180				}
   181				if let Some(slot) = current.downcast_mut::<u16>() {
   182					*slot = u16::try_from(*next).map_err(|_| InspectorEditError::IntegerOutOfRange {
   183						target_type: "u16",
   184						value: *next,
   185					})?;
   186					return Ok(());
   187				}
   188				if let Some(slot) = current.downcast_mut::<u32>() {
   189					*slot = u32::try_from(*next).map_err(|_| InspectorEditError::IntegerOutOfRange {
   190						target_type: "u32",
   191						value: *next,
   192					})?;
   193					return Ok(());
   194				}
   195				if let Some(slot) = current.downcast_mut::<u64>() {
   196					*slot = u64::try_from(*next).map_err(|_| InspectorEditError::IntegerOutOfRange {
   197						target_type: "u64",
   198						value: *next,
   199					})?;
   200					return Ok(());
   201				}
   202				if let Some(slot) = current.downcast_mut::<usize>() {
   203					*slot =
   204						usize::try_from(*next).map_err(|_| InspectorEditError::IntegerOutOfRange {
   205							target_type: "usize",
   206							value: *next,
   207						})?;
   208					return Ok(());
   209				}
   210	
   211				Err(InspectorEditError::UnsupportedValueType { actual_type })
   212			}
   213			InspectorEditValue::Float(next) => {
   214				if let Some(slot) = current.downcast_mut::<f32>() {
   215					if !next.is_finite() || *next < f32::MIN as f64 || *next > f32::MAX as f64 {
   216						return Err(InspectorEditError::FloatOutOfRange {
   217							target_type: "f32",
   218							value: *next,
   219						});
   220					}
   221					*slot = *next as f32;
   222					return Ok(());
   223				}
   224				if let Some(slot) = current.downcast_mut::<f64>() {
   225					*slot = *next;
   226					return Ok(());
   227				}
   228	
   229				Err(InspectorEditError::UnsupportedValueType { actual_type })
   230			}
   231			InspectorEditValue::Text(next) => {
   232				let slot = current
   233					.downcast_mut::<String>()
   234					.ok_or_else(|| InspectorEditError::UnsupportedValueType {
   235						actual_type: actual_type.clone(),
   236					})?;
   237				*slot = next.clone();
   238				Ok(())
   239			}
   240		}
   241	}

    14		BeginPreview,
    15		UpdatePreview,

    22			.intents
    23			.into_iter()
    24			.map(|intent| match intent {
    25				ToolIntent::Scene(scene) => ToolAction::Scene(scene),
    26				ToolIntent::SelectEntity(entity) => {

    28				}
    29				ToolIntent::ClearSelection => ToolAction::ClearSelection,
    30				ToolIntent::SetHoverEntity(entity) => ToolAction::HoverEntity(entity),
    31				ToolIntent::BeginPreview => ToolAction::BeginPreview,
    32				ToolIntent::UpdatePreview => ToolAction::UpdatePreview,
    33				ToolIntent::CommitPreview => ToolAction::CommitPreview,
    34				ToolIntent::CancelPreview => ToolAction::CancelPreview,
