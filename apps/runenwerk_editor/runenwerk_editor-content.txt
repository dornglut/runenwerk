   131		pub fn dispatch_tool_action(
   132			&mut self,
   133			action: editor_tools::ToolAction,
   134		) -> Result<(), &'static str> {
   135			match action {
   136				editor_tools::ToolAction::SelectSingle(target) => match target {
   137					editor_core::SelectionTarget::Entity(entity) => {
   138						select_single_entity(&mut self.runtime, entity)?;
   139					}
   140					editor_core::SelectionTarget::Component {
   141						entity,
   142						component_type,
   143					} => {
   144						select_single_component(&mut self.runtime, entity, component_type)?;
   145					}
   146					_ => return Err("unsupported selection target for tool action"),
   147				},
   148				editor_tools::ToolAction::ClearSelection => {
   149					clear_selection(&mut self.runtime);
   150				}
   151				editor_tools::ToolAction::Scene(intent) => {
   152					let command_id = self.runtime.allocate_command_id();
   153					let transaction_id = self.runtime.allocate_transaction_id();
   154	
   155					let _ = execute_scene_command_and_push_history(
   156						&mut self.runtime,
   157						editor_scene::scene_intent_to_command(command_id, intent),
   158						"Tool Scene Action",
   159						transaction_id,
   160					)?;
   161				}
   162				editor_tools::ToolAction::HoverEntity(entity) => {
   163					self.tool_runtime_state.set_hovered_entity(entity);
   164				}
   165				editor_tools::ToolAction::BeginPreview => {
   166					let selection = self
   167						.runtime
   168						.session()
   169						.selection()
   170						.primary()
   171						.cloned()
   172						.ok_or("cannot begin preview without a primary selection")?;
   173	
   174					self.tool_runtime_state
   175						.begin_preview(selection, crate::editor_runtime::TransformToolKind::Translate)?;
   176				}
   177				editor_tools::ToolAction::UpdatePreview => {
   178					self.tool_runtime_state.update_preview()?;
   179				}
   180				editor_tools::ToolAction::CommitPreview => {
   181					let preview = self
   182						.tool_runtime_state
   183						.commit_preview()
   184						.ok_or("no active preview session")?;
   185	
   186					Self::commit_translation_preview_into_local_transform(
   187						&mut self.runtime,
   188						preview.entity,
   189						preview.translation_delta,
   190					)?;
   191				}
   192				editor_tools::ToolAction::CancelPreview => {
   193					let _preview = self
   194						.tool_runtime_state
   195						.cancel_preview()
   196						.ok_or("no active preview session")?;
   197				}
   198			}
   199	
   200			Ok(())
   201		}

   232				.world()
   233				.get::<LocalTransform>(ecs_entity)
   234				.ok_or("entity does not have LocalTransform")?;
   235	
   236			let next_x = current.translation.x + delta.x;
   237			let next_y = current.translation.y + delta.y;
   238			let next_z = current.translation.z + delta.z;
   239	
   240			let transaction_id = runtime.allocate_transaction_id();
   241			let command_x = runtime.allocate_command_id();
   242			let command_y = runtime.allocate_command_id();
   243			let command_z = runtime.allocate_command_id();
   244	
   245			let mut commands = vec![
   246				editor_scene::scene_intent_to_command(
   247					command_x,
   248					editor_scene::SceneCommandIntent::EditComponentField {
   249						entity,
   250						component_type: local_transform_type,
   251						path: InspectorPath::root()
   252							.child_field("translation")
   253							.child_field("x"),
   254						value: InspectorEditValue::Float(next_x as f64),

   258					command_y,
   259					editor_scene::SceneCommandIntent::EditComponentField {
   260						entity,
   261						component_type: local_transform_type,
   262						path: InspectorPath::root()
   263							.child_field("translation")
   264							.child_field("y"),
   265						value: InspectorEditValue::Float(next_y as f64),
   266					},
   267				),
   268				editor_scene::scene_intent_to_command(
   269					command_z,

   271						entity,
   272						component_type: local_transform_type,
   273						path: InspectorPath::root()
   274							.child_field("translation")
   275							.child_field("z"),
   276						value: InspectorEditValue::Float(next_z as f64),
   277					},
   278				),
   279			];
   280	
   281			let _ = crate::editor_runtime::execute_scene_transaction_and_push_history(
   282				runtime,
   283				transaction_id,
   284				"Apply Translation Preview",
   285				&mut commands,
   286			)?;

   289		}

   164				Some(InspectTarget::Component {

   170						.entity_display_name(entity)

   182						.into_iter()
   183						.map(|descriptor| {
   184							let is_selected = descriptor.component_type == component_type;
   185							InspectorComponentItem::from_descriptor(descriptor, is_selected)

   257		let mut current = current;
   258	
   259		for segment in path.segments() {

   270					return Err(InspectorEditError::UnsupportedPathSegment);
   271				}
   272			}

   276			return Ok(InspectorEditValue::Bool(*value));
   277		}
   278		if let Some(value) = current.downcast_ref::<i8>() {

   281		if let Some(value) = current.downcast_ref::<i16>() {

   286		}
   287		if let Some(value) = current.downcast_ref::<i64>() {
   288			return Ok(InspectorEditValue::Integer(*value));
   289		}
   290		if let Some(value) = current.downcast_ref::<isize>() {
   291			return Ok(InspectorEditValue::Integer(*value as i64));
